/* CSS Universal Selector */
*{
    margin:0;
    padding:0;
}
 /*
 Definition:
 The * selector is called the universal selector.
 It selects all HTML elements.
*/


/* Body Selector */
body{
    background-color:#f0f0f0;
    font-family:Arial;
}
 /*
 Definition:
 The body selector is used to style the whole webpage.
*/


/* Element Selector */
h1{
    color:blue;
}
 /*
 Definition:
 Element selector styles all elements of the same tag.
*/


/* Class Selector */
.heading{
    color:red;
}
 /*
 Definition:
 Class selector is used to style elements with the same class name.
*/


/* ID Selector */
#title{
    font-size:40px;
}
 /*
 Definition:
 ID selector is used to style one unique element.
*/


/* Background Color */
.box{
    background-color:yellow;
}
 /*
 Definition:
 background-color changes the background color of an element.
*/


/* Text Color */
.text{
    color:green;
}
 /*
 Definition:
 color property changes the text color.
*/


/* Font Size */
.para{
    font-size:20px;
}
 /*
 Definition:
 font-size changes the size of text.
*/


/* Font Family */
.font{
    font-family:Verdana;
}
 /*
 Definition:
 font-family changes the style of text font.
*/


/* Font Weight */
.bold{
    font-weight:bold;
}
 /*
 Definition:
 font-weight makes text bold or light.
*/


/* Text Align */
.center{
    text-align:center;
}
 /*
 Definition:
 text-align aligns text left, right, or center.
*/


/* Border */
.border-box{
    border:2px solid black;
}
 /*
 Definition:
 border adds outline around an element.
*/


/* Margin */
.margin-box{
    margin:20px;
}
 /*
 Definition:
 margin creates space outside the element.
*/


/* Padding */
.padding-box{
    padding:20px;
}
 /*
 Definition:
 padding creates space inside the element.
*/


/* Width */
.width-box{
    width:200px;
}
 /*
 Definition:
 width sets the width of an element.
*/


/* Height */
.height-box{
    height:100px;
}
 /*
 Definition:
 height sets the height of an element.
*/


/* Display */
.display-box{
    display:flex;
}
 /*
 Definition:
 display defines how elements are displayed.
*/


/* Position */
.position-box{
    position:relative;
}
 /*
 Definition:
 position controls element placement.
*/


/* Flexbox */
.flex-container{
    display:flex;
    justify-content:center;
}
 /*
 Definition:
 Flexbox is used for flexible layouts.
*/


/* Grid */
.grid-container{
    display:grid;
    grid-template-columns:auto auto;
}
 /*
 Definition:
 Grid is used to create grid layouts.
*/


/* Hover Effect */
button:hover{
    background-color:black;
    color:white;
}
 /*
 Definition:
 :hover applies style when mouse moves over an element.
*/


/* Transition */
.transition-box{
    transition:0.5s;
}
 /*
 Definition:
 transition creates smooth animation effects.
*/


/* Transform */
.transform-box{
    transform:rotate(10deg);
}
 /*
 Definition:
 transform changes shape, size, or position of elements.
*/


/* Box Shadow */
.shadow{
    box-shadow:2px 2px 10px gray;
}
 /*
 Definition:
 box-shadow adds shadow around elements.
*/


/* Border Radius */
.round{
    border-radius:10px;
}
 /*
 Definition:
 border-radius makes corners rounded.
*/


/* Opacity */
.fade{
    opacity:0.5;
}
 /*
 Definition:
 opacity controls transparency of elements.
*/
